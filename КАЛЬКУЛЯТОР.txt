    // ========== ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ==========
    
    // Приведение числа к диапазону 1-22 (для арканов)
    function toArcane(num) {
        if (num === 0) return 22;
        let result = num;
        while (result > 22) result -= 22;
        while (result < 1) result += 22;
        return result;
    }

    // Приведение числа к однозначному (1-9)
    function toSingleDigit(num) {
        let result = num;
        while (result > 9) {
            result = result.toString().split('').reduce((sum, d) => sum + parseInt(d, 10), 0);
        }
        return result;
    }

    // Сумма цифр числа (кверсумма)
    function digitSum(num) {
        return Math.abs(num).toString().split('').reduce((sum, d) => sum + parseInt(d, 10), 0);
    }

    // Получение числа души (дата рождения -> однозначное)
    function getSoulNumber(day) {
        return toSingleDigit(day);
    }

    // Получение числа судьбы (сумма всех цифр даты -> однозначное)
    function getDestinyNumber(day, month, year) {
        const allDigits = day.toString() + month.toString() + year.toString();
        const sum = allDigits.split('').reduce((total, digit) => total + parseInt(digit, 10), 0);
        return toSingleDigit(sum);
    }

    // ========== ДВОЙНОЙ ДА (только для чисел 23-31) ==========
    // Согласно таблице:
    // 23 → основной 5, дополнительный 1
    // 24 → основной 6, дополнительный 2
    // 25 → основной 7, дополнительный 3
    // 26 → основной 8, дополнительный 4
    // 27 → основной 9, дополнительный 5
    // 28 → основной 1, дополнительный 6
    // 29 → основной 2, дополнительный 7
    // 30 → основной 3, дополнительный 8
    // 31 → основной 4, дополнительный 9
    function getDoubleDA(day) {
        if (day >= 23 && day <= 31) {
            const doubleDAMap = {
                23: { main: 5, additional: 1 },
                24: { main: 6, additional: 2 },
                25: { main: 7, additional: 3 },
                26: { main: 8, additional: 4 },
                27: { main: 9, additional: 5 },
                28: { main: 1, additional: 6 },
                29: { main: 2, additional: 7 },
                30: { main: 3, additional: 8 },
                31: { main: 4, additional: 9 }
            };
            return doubleDAMap[day];
        }
        return null;
    }

    // ========== РАСЧЁТ ПОКАЗАТЕЛЕЙ ДЛЯ КОНКРЕТНОГО ДА ==========
    // daValue — значение ДА (1-22)
    function calculateForDA(daValue, maRaw, gaRaw) {
        // СЗ (социальная задача) = ДА + МА + ГА
        let czRaw = daValue + maRaw + gaRaw;
        let cz = toArcane(czRaw);
        
        // ОПВ (ошибка прошлого воплощения) = |ДА - МА|
        let opvRaw = Math.abs(daValue - maRaw);
        let opv = toArcane(opvRaw);
        
        // ТП (точка препятствий) = ДА + МА
        let tpRaw = daValue + maRaw;
        let tp = toArcane(tpRaw);
        
        // ЗП (зона профессий) = ДА + 4×МА + ГА
        let zpRaw = daValue + 4 * maRaw + gaRaw;
        let zp = toArcane(zpRaw);
        
        // ЗЗ (зона здоровья) = ДА + 2×МА + ГА
        let zzRaw = daValue + 2 * maRaw + gaRaw;
        let zz = toArcane(zzRaw);
        
        // ЗД (зона денег) = ДА + 10×МА + ГА
        let zdRaw = daValue + 10 * maRaw + gaRaw;
        let zd = toArcane(zdRaw);
        
        // КУ1 = СЗ + ОПВ
        let ku1Raw = czRaw + opvRaw;
        let ku1 = toArcane(ku1Raw);
        
        // КУ2 = ДА + ОПВ
        let ku2Raw = daValue + opvRaw;
        let ku2 = toArcane(ku2Raw);
        
        // КУ3 = |ДА - ГА|
        let ku3Raw = Math.abs(daValue - gaRaw);
        let ku3 = toArcane(ku3Raw);
        
        // КУ4 = ГА + СЗ
        let ku4Raw = gaRaw + czRaw;
        let ku4 = toArcane(ku4Raw);
        
        return {
            da: daValue,
            cz, opv, tp, zp, zz, zd,
            ku1, ku2, ku3, ku4,
            czRaw, opvRaw, tpRaw, zpRaw, zzRaw, zdRaw
        };
    }

    // ========== ОСНОВНЫЕ НУМЕРОЛОГИЧЕСКИЕ ПОКАЗАТЕЛИ ==========
    function calculateNumerology(birthDate) {
        const [year, month, day] = birthDate.split('-').map(Number);
        
        // Число Души и Судьбы
        const soulNumber = getSoulNumber(day);
        const destinyNumber = getDestinyNumber(day, month, year);
        
        // Константы
        const maRaw = month;
        const ma = toArcane(maRaw);
        const gaRaw = digitSum(year);
        const ga = toArcane(gaRaw);
        
        // ЖА (аркан жизни)
        const allDigits = day.toString() + month.toString() + year.toString();
        const jaRaw = allDigits.split('').reduce((sum, d) => sum + parseInt(d, 10), 0);
        const ja = toArcane(jaRaw);
        
        // КБ (кармическая болезнь) = |МА - ГА|
        const kbRaw = Math.abs(maRaw - gaRaw);
        const kb = toArcane(kbRaw);
        
        // Двойной ДА
        const doubleDA = getDoubleDA(day);
        
        // Расчёт для основного ДА
        let mainResult = null;
        let additionalResult = null;
        
        if (doubleDA) {
            // Для чисел 23-31: два варианта
            mainResult = calculateForDA(doubleDA.main, maRaw, gaRaw);
            additionalResult = calculateForDA(doubleDA.additional, maRaw, gaRaw);
        } else {
            // Для обычных чисел (1-22) — один вариант
            const regularDa = toArcane(day);
            mainResult = calculateForDA(regularDa, maRaw, gaRaw);
        }
        
        // Энергия Архангела (ЭА)
        let dayStr = day.toString().padStart(2, '0');
        let monthStr = month.toString().padStart(2, '0');
        let dm = parseInt(dayStr + monthStr, 10);
        let eaProduct = dm * year;
        let eaSum = digitSum(eaProduct);
        
        // Родовая энергия (ЭР)
        let erProduct = day * month * year;
        let erSum = digitSum(erProduct);
        
        // Срединная линия
        const midline = (eaSum / 7).toFixed(2);
        
        // Программный код
        let fullNumber = parseInt(dayStr + monthStr + year.toString(), 10);
        let division = fullNumber / 7;
        let divisionStr = division.toString();
        let decimalPart = "";
        if (divisionStr.includes('.')) {
            decimalPart = divisionStr.split('.')[1];
        } else if (divisionStr.includes(',')) {
            decimalPart = divisionStr.split(',')[1];
        }
        let programCode = decimalPart.charAt(0);
        if (programCode === "" || isNaN(parseInt(programCode))) {
            programCode = 0;
        } else {
            programCode = parseInt(programCode, 10);
        }
        
        return {
            soulNumber,
            destinyNumber,
            doubleDA,
            day,
            maRaw, ma,
            gaRaw, ga,
            ja, kb,
            dm, eaProduct, eaSum, erProduct, erSum, midline, programCode,
            mainResult,
            additionalResult
        };
    }
